快速排序程序分析

package com.google.javase.zs;
//快速排序
public class QuickSore{
public static void quickSore(int[] data, int left, int right) { //该方法的递归即为分治法
int base;
if(right>left) { //if内的条件即为递归中止条件
base=sort(data,left,right);
quickSore(data,left,base-1);//小于基准
quickSore(data,base+1,right);//大于基准
}
}
public static int sort(int[] data, int left, int right) { //sort函数即为在一个基准下的排序
int zj=data[left];
int base=left,zjz; //普通快排为选取数组第一个元素的值为基准值
data[left]=data[right]; //此两步将原数组的首元素与尾元素的值互换
data[right]=zj;
for(int i=left;i<right;++i) {
if(data[right]>data[i]) { //该if语句等价于以末尾值为基准值进行快排
zjz=data[base];
data[base]=data[i];
data[i]=zjz;
base++;
}
}
zjz=data[base]; //此处zjz的值为数组中小于等于末尾值的所有值中序号最大的元素的值
data[base]=data[right];
data[right]=zjz; //此处交换data【base】和data【right】的值之后，data【base】之前的所有值均小于data【base】，其之后的所有值均大于data【base】
return base;
}
public static void main(String[] args) {
int []data= {3,4,1,2,6,4,5,7,8,0,9,43,98,88,5,7};
quickSore(data,0,data.length-1);
for(int i=0;i<data.length;++i) {
System.out.print(data[i]+” “);
}
System.out.println();
}
}
